#!/bin/bash

if [ -f $1 ]
then
    for word in $(grep -o -i -E '(^|b)S*(b|$)' $1|sort -u -f | grep -o -i -E 'w*' | sort -u -f)
    do
        word_stat=$(grep -o -i -E "(^|\W)$word($|\W)" $1 |wc -l)
        echo ""$word" $word_stat"
    done;
fi;
$ ./wordstat.sh text.txt
